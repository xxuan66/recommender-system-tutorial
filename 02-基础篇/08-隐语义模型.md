# 第八章：隐语义模型

> 通过 latent factor model 发现用户和物品的隐含关系

---

## 8.1 什么是隐语义模型

### 8.1.1 核心思想

**隐语义模型（Latent Factor Model, LFM）** 的基本思想是：

> 通过分析用户的行为数据，自动发现隐含的主题/类别，将用户和物品映射到同一隐含空间。

### 8.1.2 与矩阵分解的关系

**LFM 和矩阵分解本质相同，只是视角不同：**

| 维度 | 矩阵分解 | 隐语义模型 |
|------|---------|-----------|
| **数学视角** | 矩阵分解 | 概率模型 |
| **优化方法** | SVD/梯度下降 | EM 算法/梯度下降 |
| **解释角度** | 降维 | 隐含主题 |

### 8.1.3 直观例子

**图书推荐场景：**

```
假设有以下隐含主题：
- 主题 1：科幻小说
- 主题 2：言情小说
- 主题 3：历史传记

用户 A 的阅读偏好：[0.8, 0.1, 0.3] — 喜欢科幻
图书 X 的主题分布：[0.9, 0.05, 0.1] — 科幻小说

→ 推荐图书 X 给用户 A
```

## 8.2 LFM 算法原理

### 8.2.1 模型定义

```
预测评分：r̂ui = μ + bu + bi + Σ(k=1 to K) puk × qik

其中：
- μ: 全局平均评分
- bu: 用户 u 的偏置
- bi: 物品 i 的偏置
- puk: 用户 u 在第 k 个隐含因子上的得分
- qik: 物品 i 在第 k 个隐含因子上的得分
- K: 隐含因子数量
```

### 8.2.2 损失函数

```
最小化：L = Σ(u,i)∈R (rui - r̂ui)^2 + λ(||P||^2 + ||Q||^2 + bu^2 + bi^2)
```

### 8.2.3 梯度下降更新

```python
import numpy as np

class LFM:
    """隐语义模型实现"""
    
    def __init__(self, n_factors=10, learning_rate=0.01, regularization=0.02, n_epochs=100):
        """
        Args:
            n_factors: 隐含因子数量
            learning_rate: 学习率
            regularization: 正则化参数
            n_epochs: 训练轮数
        """
        self.n_factors = n_factors
        self.learning_rate = learning_rate
        self.regularization = regularization
        self.n_epochs = n_epochs
        
    def fit(self, ratings):
        """训练模型"""
        self.ratings = ratings
        n_users, n_items = ratings.shape
        
        # 初始化参数
        np.random.seed(42)
        self.P = np.random.normal(0, 0.1, (n_users, self.n_factors))
        self.Q = np.random.normal(0, 0.1, (n_items, self.n_factors))
        
        self.global_mean = np.mean(ratings[ratings > 0])
        self.bu = np.zeros(n_users)
        self.bi = np.zeros(n_items)
        
        # 找到所有已知评分
        known_ratings = []
        for u in range(n_users):
            for i in range(n_items):
                if ratings[u, i] > 0:
                    known_ratings.append((u, i, ratings[u, i]))
        
        # 梯度下降训练
        for epoch in range(self.n_epochs):
            np.random.shuffle(known_ratings)
            total_loss = 0
            
            for u, i, r in known_ratings:
                # 预测
                pred = self.global_mean + self.bu[u] + self.bi[i] + np.dot(self.P[u], self.Q[i])
                
                # 误差
                error = r - pred
                total_loss += error ** 2
                
                # 更新偏置
                self.bu[u] += self.learning_rate * (error - self.regularization * self.bu[u])
                self.bi[i] += self.learning_rate * (error - self.regularization * self.bi[i])
                
                # 更新隐含因子
                pu_old = self.P[u].copy()
                self.P[u] += self.learning_rate * (error * self.Q[i] - self.regularization * self.P[u])
                self.Q[i] += self.learning_rate * (error * pu_old - self.regularization * self.Q[i])
            
            # 打印进度
            if (epoch + 1) % 20 == 0:
                rmse = np.sqrt(total_loss / len(known_ratings))
                print(f"Epoch {epoch+1}/{self.n_epochs}, RMSE: {rmse:.4f}")
        
        return self
    
    def predict(self, u, i):
        """预测评分"""
        pred = self.global_mean + self.bu[u] + self.bi[i] + np.dot(self.P[u], self.Q[i])
        return np.clip(pred, 0.5, 5.0)
    
    def recommend(self, u, n=10):
        """推荐 Top-N 物品"""
        n_items = self.ratings.shape[1]
        
        predictions = []
        for i in range(n_items):
            if self.ratings[u, i] == 0:
                score = self.predict(u, i)
                predictions.append((i, score))
        
        predictions.sort(key=lambda x: x[1], reverse=True)
        return predictions[:n]
    
    def get_user_factors(self, u):
        """获取用户隐含因子"""
        return self.P[u]
    
    def get_item_factors(self, i):
        """获取物品隐含因子"""
        return self.Q[i]


# 使用示例
if __name__ == "__main__":
    ratings = np.array([
        [5, 3, 0, 4, 0],
        [4, 0, 3, 5, 2],
        [0, 4, 5, 0, 3],
        [3, 0, 4, 0, 5],
        [0, 5, 0, 3, 4],
    ])
    
    model = LFM(n_factors=5, learning_rate=0.01, regularization=0.02, n_epochs=100)
    model.fit(ratings)
    
    # 为用户 0 推荐
    user_id = 0
    recommendations = model.recommend(user_id, n=3)
    
    print(f"\n为用户 {user_id} 推荐的物品：")
    for item_id, score in recommendations:
        print(f"  物品 {item_id}: 预测评分 {score:.2f}")
    
    # 查看隐含因子
    print(f"\n用户 {user_id} 的隐含因子：{model.get_user_factors(user_id)[:5]}")
    print(f"物品 0 的隐含因子：{model.get_item_factors(0)[:5]}")
```

## 8.3 隐语义模型的应用

### 8.3.1 物品相似度计算

**基于隐含因子计算物品相似度：**

```python
def item_similarity_lfm(model, i, j):
    """
    计算物品 i 和 j 的相似度（基于隐含因子）
    """
    qi = model.get_item_factors(i)
    qj = model.get_item_factors(j)
    
    # 余弦相似度
    similarity = np.dot(qi, qj) / (np.linalg.norm(qi) * np.linalg.norm(qj))
    return similarity

# 示例
sim = item_similarity_lfm(model, 0, 1)
print(f"物品 0 和物品 1 的相似度：{sim:.4f}")
```

### 8.3.2 用户相似度计算

```python
def user_similarity_lfm(model, u, v):
    """
    计算用户 u 和 v 的相似度
    """
    pu = model.get_user_factors(u)
    pv = model.get_user_factors(v)
    
    similarity = np.dot(pu, pv) / (np.linalg.norm(pu) * np.linalg.norm(pv))
    return similarity
```

### 8.3.3 隐含因子可视化

```python
import matplotlib.pyplot as plt
from sklearn.decomposition import PCA

def visualize_factors(model, items=None, users=None):
    """
    可视化隐含因子（降维到 2D）
    """
    if items is not None:
        # 物品可视化
        factors = model.Q
        labels = [f"物品{i}" for i in range(len(factors))]
    else:
        # 用户可视化
        factors = model.P
        labels = [f"用户{u}" for u in range(len(factors))]
    
    # PCA 降维到 2D
    pca = PCA(n_components=2)
    factors_2d = pca.fit_transform(factors)
    
    # 绘图
    plt.figure(figsize=(10, 8))
    plt.scatter(factors_2d[:, 0], factors_2d[:, 1])
    
    for i, (x, y) in enumerate(factors_2d):
        plt.annotate(labels[i], (x, y), fontsize=9)
    
    plt.xlabel(f'PC1 ({pca.explained_variance_ratio_[0]:.2%})')
    plt.ylabel(f'PC2 ({pca.explained_variance_ratio_[1]:.2%})')
    plt.title('隐含因子可视化')
    plt.grid(True, alpha=0.3)
    plt.show()

# 使用
visualize_factors(model, items=True)
```

## 8.4 带偏置的 LFM

### 8.4.1 为什么需要偏置

**问题：** 有些用户习惯打高分，有些物品本身评分就高

**解决方案：** 加入用户偏置和物品偏置

```
r̂ui = μ + bu + bi + pu^T × qi
```

**说明：**
- `μ`: 全局平均分
- `bu`: 用户 u 的评分偏置（正数表示该用户习惯打高分）
- `bi`: 物品 i 的评分偏置（正数表示该物品普遍评分高）

### 8.4.2 代码实现

```python
class LFMWithBias(LFM):
    """带偏置的 LFM"""
    
    def fit(self, ratings):
        """训练模型（带偏置）"""
        self.ratings = ratings
        n_users, n_items = ratings.shape
        
        # 初始化
        np.random.seed(42)
        self.P = np.random.normal(0, 0.1, (n_users, self.n_factors))
        self.Q = np.random.normal(0, 0.1, (n_items, self.n_factors))
        
        self.global_mean = np.mean(ratings[ratings > 0])
        self.bu = np.zeros(n_users)
        self.bi = np.zeros(n_items)
        
        known_ratings = []
        for u in range(n_users):
            for i in range(n_items):
                if ratings[u, i] > 0:
                    known_ratings.append((u, i, ratings[u, i]))
        
        for epoch in range(self.n_epochs):
            np.random.shuffle(known_ratings)
            total_loss = 0
            
            for u, i, r in known_ratings:
                # 预测（带偏置）
                pred = self.global_mean + self.bu[u] + self.bi[i] + np.dot(self.P[u], self.Q[i])
                error = r - pred
                total_loss += error ** 2
                
                # 更新偏置
                self.bu[u] += self.learning_rate * (error - self.regularization * self.bu[u])
                self.bi[i] += self.learning_rate * (error - self.regularization * self.bi[i])
                
                # 更新隐含因子
                pu_old = self.P[u].copy()
                self.P[u] += self.learning_rate * (error * self.Q[i] - self.regularization * self.P[u])
                self.Q[i] += self.learning_rate * (error * pu_old - self.regularization * self.Q[i])
            
            if (epoch + 1) % 20 == 0:
                rmse = np.sqrt(total_loss / len(known_ratings))
                print(f"Epoch {epoch+1}/{self.n_epochs}, RMSE: {rmse:.4f}")
        
        return self
    
    def predict(self, u, i):
        """预测（带偏置）"""
        pred = self.global_mean + self.bu[u] + self.bi[i] + np.dot(self.P[u], self.Q[i])
        return np.clip(pred, 0.5, 5.0)
    
    def explain_prediction(self, u, i):
        """解释预测结果"""
        pred = self.predict(u, i)
        print(f"预测用户{u}对物品{i}的评分：{pred:.2f}")
        print(f"  全局平均：{self.global_mean:.2f}")
        print(f"  用户偏置：{self.bu[u]:.2f}")
        print(f"  物品偏置：{self.bi[i]:.2f}")
        print(f"  隐含因子贡献：{np.dot(self.P[u], self.Q[i]):.2f}")
```

## 8.5 实战：电影推荐

### 8.5.1 使用 MovieLens 数据集

```python
from surprise import Dataset, SVD, train_test_split, accuracy

# 加载 MovieLens 数据集
data = Dataset.load_builtin('ml-100k')

# 划分数据集
trainset, testset = train_test_split(data, test_size=0.25, random_state=42)

# 创建 LFM 模型（使用 SVD，本质相同）
model = SVD(
    n_factors=50,
    n_epochs=100,
    lr_all=0.005,
    reg_all=0.02,
    biased=True  # 使用偏置
)

# 训练
model.fit(trainset)

# 预测
predictions = model.test(testset)

# 评估
rmse = accuracy.rmse(predictions)
mae = accuracy.mae(predictions)

print(f"RMSE: {rmse:.4f}")
print(f"MAE: {mae:.4f}")
```

### 8.5.2 个性化推荐

```python
def get_top_n_recommendations(model, user_id, n=10):
    """
    获取用户的 Top-N 推荐
    """
    # 获取所有物品
    all_items = [item_id for item_id, _ in model.trainset.ir_items]
    
    # 预测所有物品评分
    predictions = []
    for item_id in all_items:
        if model.trainset.ur[user_id]:  # 检查用户是否有评分
            # 检查是否已评分
            already_rated = any(iid == item_id for (_, iid, _) in model.trainset.ur[user_id])
            if not already_rated:
                pred = model.predict(user_id, item_id)
                predictions.append((pred.iid, pred.est))
    
    # 排序
    predictions.sort(key=lambda x: x[1], reverse=True)
    return predictions[:n]

# 为用户推荐
user_id = 0
top_n = get_top_n_recommendations(model, user_id, n=10)

print(f"为用户 {user_id} 推荐的电影：")
for movie_id, score in top_n:
    print(f"  电影 {movie_id}: 预测评分 {score:.2f}")
```

## 8.7 LFM 的优缺点

### 8.7.1 优点

| 优点 | 说明 |
|------|------|
| **自动发现特征** | 无需人工标注 |
| **处理稀疏数据** | 可以处理大规模稀疏矩阵 |
| **可扩展性强** | 容易加入新特征 |
| **理论基础好** | 数学推导清晰 |

### 8.7.2 缺点

| 缺点 | 说明 | 解决方案 |
|------|------|---------|
| **隐含特征难解释** | 不知道每个因子代表什么 | 可视化分析 |
| **需要调参** | 因子数量、学习率等 | 网格搜索 |
| **训练时间长** | 大数据集慢 | 分布式训练 |
| **冷启动问题** | 新用户/物品无法处理 | 结合其他方法 |
