import pandas as pd

        ratings_df, movies_df
    """

    print(f"用户数：{ratings_df['userId'].nunique()}")
    print(f"电影数：{ratings_df['movieId'].nunique()}")
    
    return ratings_df, movies_df


def prepare_surprise_data(ratings_df):
    """
    转换为 Surprise 格式
    
    Args:
        ratings_df: 评分 DataFrame
    
    Returns:
        Surprise Dataset
    """
    reader = Reader(rating_scale=(0.5, 5.0))
    data = Dataset.load_from_df(
        ratings_df[['userId', 'movieId', 'rating']],
        reader
    )
    return data

import pandas as pd

## 16.3 推荐模型

### 16.3.1 协同过滤

```python
# models/cf.py
from surprise import KNNBasic, SVD
from surprise.model_selection import train_test_split
from surprise import accuracy
import pandas as pd

class CollaborativeFiltering:
    """协同过滤推荐"""
    
    def __init__(self, model_type='item'):
        """
        Args:
            model_type: 'user' 或 'item'
        """
        self.model_type = model_type
        
        if model_type == 'user':
            sim_options = {'name': 'pearson', 'user_based': True}
        else:
            sim_options = {'name': 'cosine', 'user_based': False}
        
        self.model = KNNBasic(
            sim_options=sim_options,
            k=50,
            min_k=5
        )
    
    def train(self, trainset):
        """训练模型"""
        self.model.fit(trainset)
        return self
    
    def predict(self, user_id, item_id):
        """预测评分"""
        return self.model.predict(user_id, item_id).est
    
    def recommend(self, user_id, all_items, n=10):
        """
        为用户推荐
        
        Args:
            user_id: 用户 ID
            all_items: 所有物品列表
            n: 推荐数量
        
        Returns:
            推荐列表 [(item_id, score), ...]
        """
        predictions = []
        
        for item_id in all_items:
            pred = self.predict(user_id, item_id)
            predictions.append((item_id, pred))
        
        predictions.sort(key=lambda x: x[1], reverse=True)
        return predictions[:n]


# 使用示例
if __name__ == "__main__":
    from surprise import Dataset
    
    data = Dataset.load_builtin('ml-100k')
    trainset, testset = train_test_split(data, test_size=0.25)
    
    model = CollaborativeFiltering(model_type='item')
    model.train(trainset)
    
    # 评估
    predictions = model.model.test(testset)
    print(f"RMSE: {accuracy.rmse(predictions):.4f}")
```

### 16.3.2 矩阵分解

```python
# models/mf.py
from surprise import SVD, NMF
from surprise.model_selection import cross_validate
import numpy as np

class MatrixFactorization:
    """矩阵分解推荐"""
    
    def __init__(
        self,
        n_factors=100,
        n_epochs=50,
        lr_all=0.005,
        reg_all=0.02
    ):
        """
        Args:
            n_factors: 隐含因子数量
            n_epochs: 训练轮数
            lr_all: 学习率
            reg_all: 正则化参数
        """
        self.model = SVD(
            n_factors=n_factors,
            n_epochs=n_epochs,
            lr_all=lr_all,
            reg_all=reg_all,
            random_state=42
        )
    
    def train(self, trainset):
        """训练模型"""
        self.model.fit(trainset)
        return self
    
    def predict(self, user_id, item_id):
        """预测评分"""
        return self.model.predict(user_id, item_id).est
    
    def get_user_factors(self, user_id):
        """获取用户隐含因子"""
        return self.model.pu[user_id]
    
    def get_item_factors(self, item_id):
        """获取物品隐含因子"""
        return self.model.qi[item_id]
    
    def similar_items(self, item_id, all_items, n=10):
        """
        查找相似物品
        
        Args:
            item_id: 物品 ID
            all_items: 所有物品
            n: 返回数量
        
        Returns:
            相似物品列表
        """
        from sklearn.metrics.pairwise import cosine_similarity
        
        item_vec = self.get_item_factors(item_id).reshape(1, -1)
        
        similarities = []
        for other_id in all_items:
            if other_id != item_id:
                other_vec = self.get_item_factors(other_id).reshape(1, -1)
                sim = cosine_similarity(item_vec, other_vec)[0][0]
                similarities.append((other_id, sim))
        
        similarities.sort(key=lambda x: x[1], reverse=True)
        return similarities[:n]


# 使用示例
if __name__ == "__main__":
    from surprise import Dataset, train_test_split
    from surprise import accuracy
    
    data = Dataset.load_builtin('ml-100k')
    trainset, testset = train_test_split(data, test_size=0.25)
    
    model = MatrixFactorization(n_factors=50, n_epochs=100)
    model.train(trainset)
    
    # 评估
    predictions = model.model.test(testset)
    print(f"RMSE: {accuracy.rmse(predictions):.4f}")
```

## 16.4 Web 应用

### 16.4.1 Flask 主程序

```python
# app.py
from flask import Flask, render_template, request, redirect, url_for
import pandas as pd
import sqlite3
from models.cf import CollaborativeFiltering
from models.mf import MatrixFactorization
from surprise import Dataset, Reader, Trainset

app = Flask(__name__)

# 全局变量
ratings_df = None
movies_df = None
model = None
trainset = None

def load_data():
    """加载数据"""
    global ratings_df, movies_df, model, trainset
    
    # 加载数据
    ratings_df = pd.read_csv('data/ml-latest-small/ratings.csv')
    movies_df = pd.read_csv('data/ml-latest-small/movies.csv')
    
    # 准备 Surprise 数据
    reader = Reader(rating_scale=(0.5, 5.0))
    data = Dataset.load_from_df(ratings_df[['userId', 'movieId', 'rating']], reader)
    trainset = data.build_full_trainset()
    
    # 训练模型
    model = MatrixFactorization(n_factors=50, n_epochs=100)
    model.train(trainset)
    
    print("数据加载完成！")


@app.route('/')
def index():
    """首页"""
    return render_template('index.html')


@app.route('/user/<int:user_id>')
def user_page(user_id):
    """用户页面"""
    # 获取用户评分过的电影
    user_ratings = ratings_df[ratings_df['userId'] == user_id]
    
    # 获取电影信息
    rated_movies = pd.merge(
        user_ratings,
        movies_df,
        on='movieId'
    )
    
    # 为用户推荐
    all_movies = movies_df['movieId'].tolist()
    recommendations = model.recommend(user_id, all_movies, n=10)
    
    # 获取推荐电影信息
    rec_movies = pd.merge(
        pd.DataFrame(recommendations, columns=['movieId', 'predicted_rating']),
        movies_df,
        on='movieId'
    )
    
    return render_template(
        'user.html',
        user_id=user_id,
        rated_movies=rated_movies,
        recommendations=rec_movies
    )


@app.route('/movie/<int:movie_id>')
def movie_page(movie_id):
    """电影页面"""
    movie = movies_df[movies_df['movieId'] == movie_id].iloc[0]
    
    # 获取相似电影
    all_movies = movies_df['movieId'].tolist()
    similar = model.similar_items(movie_id, all_movies, n=10)
    
    similar_movies = pd.merge(
        pd.DataFrame(similar, columns=['movieId', 'similarity']),
        movies_df,
        on='movieId'
    )
    
    return render_template(
        'movie.html',
        movie=movie,
        similar_movies=similar_movies
    )


if __name__ == '__main__':
    load_data()
    app.run(debug=True, host='0.0.0.0', port=5000)
```
